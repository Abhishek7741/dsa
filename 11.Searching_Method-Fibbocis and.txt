#include <iostream>
using namespace std;

// ---------------------- Binary Search ----------------------
int binarySearch(int arr[], int n, int key) {
    int low = 0, high = n - 1;

    while (low <= high) {
        int mid = (low + high) / 2;

        if (arr[mid] == key)
            return mid;
        else if (arr[mid] < key)
            low = mid + 1;
        else
            high = mid - 1;
    }

    return -1;  // Not found
}

// ---------------------- Fibonacci Search ----------------------
int fibonacciSearch(int arr[], int n, int key) {
    int fib2 = 0;    // F(k-2)
    int fib1 = 1;    // F(k-1)
    int fib = fib1 + fib2;  // F(k)

    while (fib < n) {
        fib2 = fib1;
        fib1 = fib;
        fib = fib1 + fib2;
    }

    int offset = -1;

    while (fib > 1) {
        int i = min(offset + fib2, n - 1);

        if (arr[i] < key) {
            fib = fib1;
            fib1 = fib2;
            fib2 = fib - fib1;
            offset = i;
        }
        else if (arr[i] > key) {
            fib = fib2;
            fib1 = fib1 - fib2;
            fib2 = fib - fib1;
        }
        else {
            return i;
        }
    }

    if (fib1 == 1 && arr[offset + 1] == key)
        return offset + 1;

    return -1;
}

// ---------------------- Main Program ----------------------
int main() {
    int arr[100], n, key, choice;

    cout << "Enter number of elements: ";
    cin >> n;

    cout << "Enter elements in sorted order:\n";
    for (int i = 0; i < n; i++)
        cin >> arr[i];

    do {
        cout << "\n--- Search Menu ---\n";
        cout << "1. Binary Search\n";
        cout << "2. Fibonacci Search\n";
        cout << "3. Exit\n";
        cout << "Enter your choice: ";
        cin >> choice;

        switch (choice) {
            case 1:
                cout << "Enter key to search: ";
                cin >> key;
                int bIndex;
                bIndex = binarySearch(arr, n, key);
                if (bIndex != -1)
                    cout << "Element found at index " << bIndex << endl;
                else
                    cout << "Element not found\n";
                break;

            case 2:
                cout << "Enter key to search: ";
                cin >> key;
                int fIndex;
                fIndex = fibonacciSearch(arr, n, key);
                if (fIndex != -1)
                    cout << "Element found at index " << fIndex << endl;
                else
                    cout << "Element not found\n";
                break;

            case 3:
                cout << "Exiting program.\n";
                break;

            default:
                cout << "Invalid choice!\n";
        }

    } while (choice != 3);

    return 0;
}
