#include <iostream>
using namespace std;

int main() {
    int amount;
    cout << "Enter amount: ";
    cin >> amount;

    int notes[] = {2000, 500, 200, 100, 50, 20, 10, 5, 2, 1};
    int n = 10;

    int count = 0;

    cout << "\nNotes used:\n";

    for (int i = 0; i < n; i++) {
        if (amount >= notes[i]) {
            int num = amount / notes[i];
            count += num;
            amount -= num * notes[i];

            cout << notes[i] << " x " << num << endl;
        }
    }

    cout << "\nMinimum notes required = " << count << endl;

    return 0;
}
