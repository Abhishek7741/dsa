#include <iostream>
using namespace std;

#define MAX 100

int queue[MAX];
int front = -1, rear = -1;

// Check if queue is empty
bool isEmpty() {
    return front == -1;
}

// Check if queue is full
bool isFull() {
    return rear == MAX - 1;
}

// Enqueue operation
void enqueue(int val) {
    if (isFull()) {
        cout << "Queue Overflow! Cannot insert " << val << endl;
        return;
    }

    if (isEmpty()) front = 0; // first element

    rear++;
    queue[rear] = val;
    cout << val << " enqueued.\n";
}

// Dequeue operation
void dequeue() {
    if (isEmpty()) {
        cout << "Queue Underflow! Cannot dequeue.\n";
        return;
    }

    cout << queue[front] << " dequeued.\n";

    if (front == rear) { // last element removed
        front = rear = -1;
    } else {
        front++;
    }
}

// Peek front element
void peek() {
    if (isEmpty()) {
        cout << "Queue is empty.\n";
        return;
    }
    cout << "Front element: " << queue[front] << endl;
}

// Display all elements
void display() {
    if (isEmpty()) {
        cout << "Queue is empty.\n";
        return;
    }

    cout << "Queue elements: ";
    for (int i = front; i <= rear; i++)
        cout << queue[i] << " ";
    cout << endl;
}

int main() {
    int choice, val;

    while (true) {
        cout << "\n--- Linear Queue Menu ---\n";
        cout << "1. Enqueue\n";
        cout << "2. Dequeue\n";
        cout << "3. Peek\n";
        cout << "4. Display\n";
        cout << "5. Exit\n";
        cout << "Enter your choice: ";
        cin >> choice;

        switch (choice) {
            case 1:
                cout << "Enter value to enqueue: ";
                cin >> val;
                enqueue(val);
                break;

            case 2:
                dequeue();
                break;

            case 3:
                peek();
                break;

            case 4:
                display();
                break;

            case 5:
                cout << "Exiting...\n";
                return 0;

            default:
                cout << "Invalid choice!\n";
        }
    }

    return 0;
}
