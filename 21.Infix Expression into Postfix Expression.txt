#include <iostream>
#include <stack>
#include <cctype>
using namespace std;

// Function to check operator precedence
int precedence(char op) {
    if (op == '^') return 3;
    if (op == '*' || op == '/') return 2;
    if (op == '+' || op == '-') return 1;
    return 0;
}

// Check if character is an operator
bool isOperator(char ch) {
    return (ch == '+' || ch == '-' || ch == '*' || ch == '/' || ch == '^');
}

// Function to convert infix to postfix
string infixToPostfix(string infix) {
    stack<char> st;
    string postfix = "";
    
    for (int i = 0; i < infix.length(); i++) {

        // Skip spaces
        if (infix[i] == ' ') continue;

        // If digit → read full multi-digit number
        if (isdigit(infix[i])) {
            while (i < infix.length() && isdigit(infix[i])) {
                postfix += infix[i];
                i++;
            }
            postfix += ' ';  // separator
            i--;
        }

        // If opening bracket
        else if (infix[i] == '(') {
            st.push(infix[i]);
        }

        // If closing bracket → pop till '('
        else if (infix[i] == ')') {
            while (!st.empty() && st.top() != '(') {
                postfix += st.top();
                postfix += ' ';
                st.pop();
            }
            if (!st.empty()) st.pop();  // pop '('
            else {
                cout << "Error: Unmatched ')'\n";
                return "";
            }
        }

        // If operator
        else if (isOperator(infix[i])) {
            while (!st.empty() && precedence(st.top()) >= precedence(infix[i])) {
                postfix += st.top();
                postfix += ' ';
                st.pop();
            }
            st.push(infix[i]);
        }

        // Invalid character
        else {
            cout << "Error: Invalid character '" << infix[i] << "'\n";
            return "";
        }
    }

    // Pop remaining operators
    while (!st.empty()) {
        if (st.top() == '(') {
            cout << "Error: Unmatched '('\n";
            return "";
        }
        postfix += st.top();
        postfix += ' ';
        st.pop();
    }

    return postfix;
}

int main() {
    string infix;
    cout << "Enter Infix Expression: ";
    getline(cin, infix);

    string postfix = infixToPostfix(infix);

    if (postfix != "")
        cout << "Postfix Expression: " << postfix << endl;

    return 0;
}
