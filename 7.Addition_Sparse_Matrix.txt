#include <iostream>
using namespace std;

#define MAX 50

void readSparse(int tuple[MAX][3]) {
    int r, c, n;

    cout << "\nEnter rows, columns and number of non-zero elements: ";
    cin >> r >> c >> n;

    tuple[0][0] = r;
    tuple[0][1] = c;
    tuple[0][2] = n;

    cout << "Enter row, column, value for non-zero elements:\n";
    for (int i = 1; i <= n; i++) {
        cin >> tuple[i][0] >> tuple[i][1] >> tuple[i][2];
    }
}

void printSparse(int tuple[MAX][3]) {
    int n = tuple[0][2];
    cout << "\nRow\tCol\tVal\n";
    for (int i = 0; i <= n; i++) {
        cout << tuple[i][0] << "\t" << tuple[i][1] << "\t" << tuple[i][2] << endl;
    }
}

void addSparse(int A[MAX][3], int B[MAX][3], int C[MAX][3]) {
    int i = 1, j = 1, k = 1;

    if (A[0][0] != B[0][0] || A[0][1] != B[0][1]) {
        cout << "\nAddition not possible: dimension mismatch.";
        return;
    }

    C[0][0] = A[0][0];
    C[0][1] = A[0][1];

    while (i <= A[0][2] && j <= B[0][2]) {

        if (A[i][0] < B[j][0] ||
           (A[i][0] == B[j][0] && A[i][1] < B[j][1])) {
            C[k][0] = A[i][0];
            C[k][1] = A[i][1];
            C[k][2] = A[i][2];
            i++; k++;
        }
        else if (B[j][0] < A[i][0] ||
                (A[i][0] == B[j][0] && B[j][1] < A[i][1])) {
            C[k][0] = B[j][0];
            C[k][1] = B[j][1];
            C[k][2] = B[j][2];
            j++; k++;
        }
        else {  
            int sum = A[i][2] + B[j][2];
            if (sum != 0) {
                C[k][0] = A[i][0];
                C[k][1] = A[i][1];
                C[k][2] = sum;
                k++;
            }
            i++; 
            j++;
        }
    }

    while (i <= A[0][2]) {
        C[k][0] = A[i][0];
        C[k][1] = A[i][1];
        C[k][2] = A[i][2];
        i++; k++;
    }

    while (j <= B[0][2]) {
        C[k][0] = B[j][0];
        C[k][1] = B[j][1];
        C[k][2] = B[j][2];
        j++; k++;
    }

    C[0][2] = k - 1;
}

int main() {
    int A[MAX][3], B[MAX][3], C[MAX][3];

    cout << "\nEnter Sparse Matrix A:";
    readSparse(A);

    cout << "\nEnter Sparse Matrix B:";
    readSparse(B);

    addSparse(A, B, C);

    cout << "\nMatrix A (3-tuple format):";
    printSparse(A);

    cout << "\nMatrix B (3-tuple format):";
    printSparse(B);

    cout << "\nResultant Matrix (A + B):";
    printSparse(C);

    return 0;
}
